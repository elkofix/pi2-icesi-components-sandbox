// 1. Importar y configurar el hook
import { useForm, SubmitHandler } from 'react-hook-form';

type FormValues = {
  username: string;
  password_rhf: string;
};

const MyFormComponent = () => {
  const { register, handleSubmit, formState: { errors }, reset } = useForm<FormValues>();

  // 2. Crear la función que maneja el envío
  const onFormSubmit: SubmitHandler<FormValues> = (data) => {
    console.log("Valores recolectados:", data);
    alert(JSON.stringify(data));
  };

  // 3. Crear función para un onClick simple
  const handleCancelClick = () => {
    reset(); // Limpia el formulario
    alert("Cancelado");
  };

  return (
    // 4. Envolver el envío con handleSubmit
    <form onSubmit={handleSubmit(onFormSubmit)}>
      <CustomInput
        id="username"
        label="Nombre de Usuario"
        // 5. Registrar el campo y añadir reglas de validación
        {...register("username", {
          required: "Este campo es obligatorio",
        })}
        // 6. Pasar el mensaje de error al componente
        error={errors.username?.message}
      />
      
      {/* ... otro input para la contraseña ... */}
      
      <CustomButton
        type="button"
        variant="secondary"
        onClick={handleCancelClick} // onClick se usa directamente
      >
        Cancelar
      </CustomButton>
      <CustomButton type="submit">
        Acceder
      </CustomButton>
    </form>
  );
};

Patrón Avanzado: Búsqueda en Tiempo Real (Backend)

Explicación de Alto Nivel

Este ejemplo es teórico y no realiza llamadas de red reales. El snippet de código de la derecha ilustra el patrón de "debouncing", que es esencial para evitar sobrecargar tu servidor.

  1. El usuario escribe en el input del `CustomSearchCard`.
  2. El estado (`searchTerm`) se actualiza al instante, pero no se hace nada más.
  3. Un hook `useDebounce` espera a que el usuario haga una pausa (ej. 500ms).
  4. Solo entonces, el valor "debounced" se actualiza.
  5. Un `useEffect` que observa este valor "debounced" se activa y realiza la llamada a la API.
  6. La tabla se actualiza con los resultados del servidor.
/*
  * Búsqueda en el Backend (Debouncing)
  *
  * Problema: Llamar a la API en cada pulsación de tecla ('a', 'an', 'ana')
  * genera tráfico de red innecesario y sobrecarga el servidor.
  *
  * Solución: Usar "debouncing". Se espera a que el usuario deje de teclear
  * por un momento (ej. 300-500ms) antes de enviar la petición a la API.
  */
  
// Hook personalizado para el debouncing (un patrón muy común en React)
function useDebounce(value, delay) {
  const [debouncedValue, setDebouncedValue] = useState(value);

  useEffect(() => {
    // Configura un temporizador para actualizar el valor después del 'delay'
    const handler = setTimeout(() => {
      setDebouncedValue(value);
    }, delay);

    // Limpia el temporizador si el valor cambia (ej. el usuario sigue escribiendo)
    return () => clearTimeout(handler);
  }, [value, delay]);

  return debouncedValue;
}

// --- En tu componente ---
const [searchTerm, setSearchTerm] = useState('');
const [results, setResults] = useState([]);
const [isLoading, setIsLoading] = useState(false);

// El valor 'debouncedSearchTerm' solo se actualizará 500ms después
// de que el usuario haya dejado de escribir en el input.
const debouncedSearchTerm = useDebounce(searchTerm, 500);

useEffect(() => {
  // Este efecto se ejecuta solo cuando el valor 'debounced' cambia.
  if (debouncedSearchTerm) {
    setIsLoading(true);
    // Lógica para llamar a tu API
    fetch(`/api/users?search=${debouncedSearchTerm}`)
      .then(res => res.json())
      .then(data => {
        setResults(data);
        setIsLoading(false);
      });
  } else {
    // Opcional: limpiar resultados si la búsqueda está vacía
    setResults([]);
  }
}, [debouncedSearchTerm]); // <-- Dependencia clave

return (
  <>
    <CustomSearchCard
      placeholder="Buscar en el servidor..."
      // El input actualiza el estado 'searchTerm' inmediatamente
      onSearchChange={setSearchTerm}
    />
    {/* La tabla se renderiza con los datos de la API */}
    {isLoading 
      ? <p>Cargando...</p> 
      : <CustomTable data={results} ... />
    }
  </>
);
